int lowerBound(vector<int> &nums, int target) {
  int l = 0;
  int r = nums.size();

  while (l < r) {
    int mid = l + (r - l) / 2;
    if (nums[mid] < target) {
      l = mid + 1;
    } else {
      r = mid;
    }
  }

  return l;
}

Diff Array

vector<int> diffArray(vector<vector<int>> &nums, vector<int> &query) {
  map<int, int> diff;
  diff[0] = 0;
  for (auto &e : nums) {
    int start = e[0], end = e[1] + 1;
    diff[start]++;
    diff[end]--;
  }

  vector<int> pos;
  vector<int> prefix;

  int curr = 0;
  for (auto pair : diff) {
    pos.push_back(pair.first);
    curr += pair.second;
    prefix.push_back(curr);
  }

  vector<int> ans;
  for (int q : query) {
    int i = upper_bound(pos.begin(), pos.end(), q) - pos.begin() - 1;
    ans.push_back(prefix[i]);
  }

  return ans;
}

int knapsack(vector<int> &weight, vector<int> &value, int capacity) {
  int n = weight.size();
  vector<vector<int>> dp(n + 1, vector<int>(capacity + 1, 0));

  for (int i = 1; i <= n; i++) {
    for (int w = 1; w <= capacity; w++) {
      if (weight[i-1] <= w) {
        dp[i][w] = max(dp[i-1][w], dp[i-1][w - weight[i-1]] + value[i-1]);
      } else {
        dp[i][w] = dp[i-1][w];
      }
    }
  }
  return dp[n][capacity];
}

void bfs(vector<vector<int>> &graph) {
  int n = graph.size();
  queue<int> q;
  vector<bool> vis(n, false);
  q.push(start);
  vis[start] = true;

  while (!q.empty()) {
    int c = q.front();
    q.pop();
    for (int adj : graph[c]) {
      if (!vis[adj]) {
        q.push(adj);
        vis[adj] = true;
      }
    }
  }
}

int dijkstra(int n, int start, int end, vector<vector<int>> &edges) {
  // build graph
  vector<vector<pair<int, int>>> graph(n);
  for (auto &edge : edges) {
    int from = edge[0], to = edge[1], cost = edge[2];
    graph[from].push_back(make_pair(to, cost));
  }


  vector<int> dist(n, INT_MAX);
  priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;
  pq.push({0, start}); // cost, node

  while (!pq.empty()) {
    int curr_cost = pq.top()[0];
    int curr_node = pq.top()[1];
    pq.pop();

    if (curr_cost > dist[curr_node]) {
      continue;
    }
    if (curr_node == end) {
      return curr_cost;
    }

    for (auto &neighbor : graph[curr_node]) {
      int next_node = neighbor.first;
      int next_cost = curr_cost + neighbor.second;

      if (next_cost < dist[next_node]) {
        dist[next_node] = next_cost;
        pq.push({next_cost, next_node});
      }
    }
  }

  return -1;
}

void floyd(int n, vector<vector<int>> &edges) {
  vector<vector<long>> matrix(n, vector<long>(n, INT_MAX));
  for (int i = 0; i < n; i++) {
    matrix[i][i] = 0;
  }

  for (auto &edge : edges) {
    int from = edge[0], to = edge[1], cost = edge[2];
    matrix[from][to] = cost;
  }

  for (int k = 0; k < n; k++) {
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {
        long tmp = matrix[i][k] + matrix[k][j];
        matrix[i][j] = min(matrix[i][j], tmp);
      }
    }
  }
}

vector<int> topologicalSort(int n, vector<vector<int>> edges) {
  vector<vector<int>> graph(n, vector<int>(n));
  vector<int> indegree(n);
  for (auto &edge : edges) {
    int u = edge[0], v = edge[1];
    graph[u].emplace_back(v);
    indegree[v]++;
  }

  queue<int> Q;
  vector<int> ret;
  for (int i = 0; i < n; i++) {
    if (indegree[i] == 0) {
      Q.push(i);
      ret.push_back(i);
    }
  }

  while (!Q.empty()) {
    int size = Q.size();
    for (int i = 0; i < size; i++) {
      int u = Q.front();
      Q.pop();
      for (int v : graph[u]) {
        if (--indegree[v] == 0) {
          Q.push(v);
          ret.push_back(v);
        }
      }
    }
  }

  return ret;
}

class UnionFind {
private:
  vector<int> parent; // Parent array to represent sets
  vector<int> rank;   // Rank to optimize union by rank

public:
  UnionFind(int n) {
    parent.resize(n);
    rank.resize(n, 0);
    for (int i = 0; i < n; i++) {
      parent[i] = i; // Initialize each element as a separate set
    }
  }

  int find(int x) {
    if (parent[x] == x) {
      return x;
    }
    return parent[x] = find(parent[x]); // Path compression
  }

  bool unionSets(int x, int y) {
    int rootX = find(x);
    int rootY = find(y);

    if (rootX == rootY) {
      return false; // Elements are already in the same set
    }

    // Union by rank to keep the tree flat
    if (rank[rootX] < rank[rootY]) {
      parent[rootX] = rootY;
    } else if (rank[rootX] > rank[rootY]) {
      parent[rootY] = rootX;
    } else {
      parent[rootY] = rootX;
      rank[rootX]++;
    }

    return true; // Elements were successfully united
  }
};

class MatrixSum {
private:
  vector<vector<int>> sum;

public:
  MatrixSum(vector<string> &maxtrix) {
    int n = maxtrix.size();
    int m = maxtrix[0].size();
    sum = vector<vector<int>>(n + 1, vector(m + 1, 0));

    for (int i = 0; i < n; i++) {
      for (int j = 0; j < m; j++) {
        sum[i+1][j+1] = sum[i+1][j] + sum[i][j+1] - sum[i][j] + (maxtrix[i][j]&1);
      }
    }
  }

  /*   返回左上角在 (r1,c1) 右下角在 (r2-1,c2-1) 的子矩阵元素和   */
  int query(int r1, int c1, int r2, int c2) {
    return sum[r2][c2] - sum[r2][c1] - sum[r1][c2] + sum[r1][c1];
  }
};

void monotonicStack(vector<int> &nums) {
  stack<int> st;
  for (int num : nums) {
    // if you want to keep a monotonically increasing stack
    // replace `num < st.top()` with `num > st.top()`
    while (!st.empty() && num < st.top()) {
      st.pop();
    }

    // since num is greater or equal to st.top()
    // stack structure is first in last out
    // so, we will get the greatest num at the top
    st.push(num);
  }
}

String

LCS (The Longest Common Subsequence)

  • LC 712. Minimum ASCII Delete Sum for Two Strings
int LCS(string s1, string s2) {
  int n = s1.size();
  int m = s2.size();
  vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
  for (int i = 1; i <= s1.size(); i++) {
    for (int j = 1; j <= s2.size(); j++) {
      if (s1[i-1] == s2[j-1]) {
        dp[i][j] = dp[i-1][j-1] + 1;
      } else {
        dp[i][j] = min(dp[i-1][j], dp[i][j-1]) + 1;
      }
    }
  }
  return dp[n][m];
}

LCA (Lowest Common Ancestor)

  • LC 1483. Kth Ancestor of a Tree Node
  • LC 2846. Minimum Edge Weight Equilibrium Queries in a Tree
class TreeAncestor {
private:
  vector<int> depth;
  vector<vector<int>> pa;

public:
  TreeAncestor(vector<pair<int, int>> &edges) {
    int n = edges.size();
    int m = 32 - __builtin_clz(n); // the length of binary n
    vector<vector<int>> g(n);
    for (auto [a, b] : edges) {
      g[a].push_back(b);
      g[b].push_back(a);
    }

    depth.resize(n);
    pa.resize(n, vector<int>(m, -1));
    function<void(int, int)> dfs = [&](int x, int parent) -> void {
      pa[x][0] = parent;
      for (int next : g[x]) {
        if (next != parent) {
          depth[next] = depth[x] + 1;
          dfs(next, x);
        }
      }
    };
    dfs(0, -1);

    for (int i = 0; i < m - 1; i++) {
      for (int x = 0; x < n; x++) {
        int p = pa[x][i];
        if (p != -1) pa[x][i+1] = pa[p][i];
      }
    }
  }

  int get_kth_ancestor(int node, int k) {
    while (k > 0) {
      node = pa[node][__builtin_clz(k)];
      k &= k - 1; // make the lowbit set to 0
    }
    return node;
  }

  int get_cla(int x, int y) {
    if (depth[x] > depth[y]) swap(x, y);

    y = get_kth_ancestor(y, depth[y] - depth[x]); // make x and y in the same depth
    if (x == y) return x;

    for (int i = pa[x].size() - 1; i >= 0; i--) {
      int px = pa[x][i], py = pa[y][i];
      if (px != py) {
        x = px;
        y = py;
      }
    }

    return pa[x][0];
  }
};

#include <iostream>
#include <unordered_map>

class TrieNode {
public:
  bool is_word;
  std::unordered_map<char, TrieNode *> children;
  TrieNode() : is_word(false) {}
};

class Trie {
private:
  TrieNode *root;
  void deleteTrie(TrieNode *node) {
    if (node) {
      for (auto &kv : node->children) {
        deleteTrie(kv.second);
      }
      delete node;
    }
  }

public:
  Trie() {
    root = new TrieNode();
  }

  ~Trie() {
    deleteTrie(root);
  }

  void insert(const std::string &word) {
    TrieNode *current = root;
    for (char c : word) {
      if (current->children.find(c) == current->children.end()) {
        current->children[c] = new TrieNode();
      }
      current = current->children[c];
    }
    current->is_word = true;
  }

  bool search(const std::string &word) {
    TrieNode *current = root;
    for (char c : word) {
      if (current->children.find(c) == current->children.end()) {
        return false;
      }
      current = current->children[c];
    }
    return current->is_word;
  }

  bool start_with(const std::string &prefix) {
    TrieNode *current = root;
    for (char c : prefix) {
      if (current->children.find(c) == current->children.end()) {
        return false;
      }
      current = current->children[c];
    }
    return true;
  }
};
